var startTime = Date.now(); function render() { var elapsed = (Date.now() - startTime) / 1000; time = (elapsed % 20) / 20; ctx.fillStyle = '#0a0a12'; ctx.fillRect(0, 0, W, H); ctx.save(); ctx.translate(W / 2, H / 2); var t1 = constrain(map(time, 0, CHANGE_EVENT + 0.25, 0, 1), 0, 1); var t2 = constrain(map(time, CHANGE_EVENT, 1, 0, 1), 0, 1); // Camera rotation ctx.rotate(-Math.PI * ease(t2, 2.7)); // Draw trail for (var i = 0; i < TRAIL_LENGTH; i++) { var f = map(i, 0, TRAIL_LENGTH, 1.1, 0.1); var sw = (1.3 * (1 - t1) + 3.0 * Math.sin(Math.PI * t1)) * f; ctx.fillStyle = 'rgba(200,180,255,' + (f * 0.3) + ')'; ctx.lineWidth = sw; var pathTime = t1 - 0.00015 * i; var pos = spiralPath(pathTime); var basePos = pos; var offset = { x: pos.x + 5, y: pos.y + 5 }; var rotated = rotatePoint(basePos, offset, t1, false); var x = rotated.x * (1 - t1) + pos.x * t1; var y = rotated.y * (1 - t1) + pos.y * t1; ctx.beginPath(); ctx.arc(x, y, sw * 0.5, 0, Math.PI * 2); ctx.fill(); } // Draw stars for (var i = 0; i < stars.length; i++) { var s = stars[i]; var t = t1 * s.speed * 1000 + s.offset; var angle = t * 0.5 + s.offset; var r = s.radius * (1 + t * 0.01); var sx = Math.cos(angle) * r; var sy = Math.sin(angle) * r * 0.8; var scale = ZOOM / (s.z + ZOOM); var px = sx * scale; var py = sy * scale; var sz = (s.z + ZOOM) / 5000; var size = Math.max(0.3, sz * 2.5); ctx.beginPath(); ctx.arc(px, py, size, 0, Math.PI * 2); ctx.fillStyle = 'rgba(' + s.color.r + ',' + s.color.g + ',' + s.color.b + ',' + (0.3 + sz * 0.5) + ')'; ctx.fill(); s.z += s.speed * 4; if (s.z > s.maxZ) { s.z = rnd(200, 1000); s.x = rnd(-1, 1) * 1000; s.y = rnd(-1, 1) * 1000; } } // Draw start dot if (time > CHANGE_EVENT) { var dy = CAMERA_Z * START_DOT_Y / ZOOM; projectDot(0, dy, TRAVEL_DIST, 2.5, t2); } ctx.restore(); requestAnimationFrame(render); } render();
